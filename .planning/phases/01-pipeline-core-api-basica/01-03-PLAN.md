```python
import asyncio
import time
from dataclasses import dataclass, field
from collections import deque
from datetime import datetime, timezone
from typing import Callable, Any
import json
import logging

logger = logging.getLogger(__name__)

class QueueFullError(Exception):
    """Cola llena — 503."""
    pass

class QueueTimeoutError(Exception):
    """Timeout esperando en cola — 504."""
    pass

@dataclass
class JobRecord:
    article_id: str
    status: str                    # "completed" | "error"
    processing_time_ms: int
    model_used: str
    timestamp: str                 # ISO 8601
    error: str | None = None

@dataclass
class QueueState:
    active_jobs: int = 0
    queued_jobs: int = 0
    total_processed: int = 0
    total_errors: int = 0
    job_history: deque = field(default_factory=lambda: deque(maxlen=50))

class JobQueue:
    def __init__(self, max_concurrent: int = 1, max_queue_size: int = 10, timeout_seconds: int = 120):
        self._semaphore = asyncio.Semaphore(max_concurrent)
        self._max_queue_size = max_queue_size
        self._timeout_seconds = timeout_seconds
        self._state = QueueState()
        self._max_concurrent = max_concurrent

    @property
    def state(self) -> QueueState:
        return self._state

    @property
    def max_concurrent(self) -> int:
        return self._max_concurrent

    async def submit_job(
        self,
        process_fn: Callable,      # Funcion sincrona (process_image)
        image_bytes: bytes,
        article_id: str,
        config_snapshot,            # AppConfig snapshot (ya tomado antes de llamar)
        rembg_session,
    ):
        """
        Encola y ejecuta un job de procesamiento.
        Raises QueueFullError si la cola esta llena.
        Raises QueueTimeoutError si el timeout expira esperando.
        """
        # Check 1: Cola llena -> 503 inmediato (per QUEUE-02)
        if self._state.queued_jobs >= self._max_queue_size:
            logger.warning(json.dumps({
                "level": "warning",
                "event": "queue_full",
                "article_id": article_id,
                "queued_jobs": self._state.queued_jobs,
                "max_queue_size": self._max_queue_size,
            }))
            raise QueueFullError(f"Queue full: {self._state.queued_jobs}/{self._max_queue_size}")

        # Incrementar contador de encolados
        self._state.queued_jobs += 1

        try:
            # Check 2: Esperar semaphore con timeout (per QUEUE-03)
            try:
                await asyncio.wait_for(
                    self._semaphore.acquire(),
                    timeout=self._timeout_seconds,
                )
            except asyncio.TimeoutError:
                self._state.queued_jobs -= 1
                logger.warning(json.dumps({
                    "level": "warning",
                    "event": "queue_timeout",
                    "article_id": article_id,
                    "timeout_seconds": self._timeout_seconds,
                }))
                raise QueueTimeoutError(
                    f"Timeout after {self._timeout_seconds}s waiting in queue"
                )

            # Semaphore adquirido: mover de queued a active
            self._state.queued_jobs -= 1
            self._state.active_jobs += 1

            start = time.monotonic()
            try:
                # CRITICO: ejecutar en thread para no bloquear event loop (per QUEUE-04)
                result = await asyncio.to_thread(
                    process_fn,
                    image_bytes,
                    article_id,
                    config_snapshot,
                    rembg_session,
                )

                elapsed_ms = int((time.monotonic() - start) * 1000)

                # Registrar job exitoso
                self._state.total_processed += 1
                self._state.job_history.append(JobRecord(
                    article_id=article_id,
                    status="completed",
                    processing_time_ms=elapsed_ms,
                    model_used=config_snapshot.rembg.model,
                    timestamp=datetime.now(timezone.utc).isoformat(),
                ))

                logger.info(json.dumps({
                    "level": "info",
                    "event": "job_complete",
                    "article_id": article_id,
                    "processing_time_ms": elapsed_ms,
                }))

                return result

            except Exception as e:
                elapsed_ms = int((time.monotonic() - start) * 1000)
                self._state.total_errors += 1
                self._state.job_history.append(JobRecord(
                    article_id=article_id,
                    status="error",
                    processing_time_ms=elapsed_ms,
                    model_used=config_snapshot.rembg.model,
                    timestamp=datetime.now(timezone.utc).isoformat(),
                    error=str(e),
                ))
                raise

            finally:
                self._state.active_jobs -= 1
                self._semaphore.release()

        except (QueueFullError, QueueTimeoutError):
            raise
        except Exception:
            # queued_jobs ya fue decrementado al adquirir semaphore
            raise
```
